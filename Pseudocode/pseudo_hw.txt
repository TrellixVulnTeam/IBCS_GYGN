BIN = [0,0,0,0,0,0,0,0]

input NUM
COUNTER=0
loop while NUM>=1
	MODDER=NUM mod 2
	NUM = (NUM-MODDER) /2
	BIN[COUNTER]=MODDER
	COUNTER=COUNTER+1
end loop

N = 8
K = 0

loop while K < (N - 1)/2
    TEMP = BIN[K]
    BIN[K] = BIN[N - K - 1]
    BIN[N - K - 1] = TEMP
    K = K + 1
end loop
	
loop C from 0 to N-1
    output BIN[C]
end loop

N=input("Enter an integer greater than zero")
K=N*N
Z=1
TOP = 0
BOTTOM = 4
LEFT = 0
RIGHT = 4
ROTATE=0
loop while Z<=K
	if ROTATE mod 4=0 then
		loop for N=LEFT to RIGHT-1
			A[TOP][N]=Z 
			Z=Z+1
		end loop
		TOP=TOP+1
	elif ROTATE mod 4=1 then
		loop for N=TOP to BOTTOM-1
			A[RIGHT][N]=Z
			Z=Z+1
		end loop
		RIGHT=RIGHT-1
	elif ROTATE mod 4=2 then
		loop for N=RIGHT to LEFT
			A[BOTTOM][N]=Z
			Z=Z+1
		end loop
		BOTTOM=BOTTOM-1
	else
		loop for N=BOTTOM to TOP
			A[LEFT][N]=Z
			Z=Z+1
		end loop
		LEFT=LEFT+1
	endif
end loop
